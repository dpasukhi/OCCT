cmake_minimum_required(VERSION 3.12)
project(TestVideoRecording)

# Find OpenCascade
find_package(OpenCASCADE REQUIRED)

# Create executable
add_executable(test_video_recording test_video_recording.cpp)

# Include directories
target_include_directories(test_video_recording PRIVATE 
    ${OpenCASCADE_INCLUDE_DIR}
    "build/inc"  # For FFmpeg_Compatibility.hxx
)

# Link libraries
target_link_libraries(test_video_recording 
    ${OpenCASCADE_LIBRARIES}
)

# Set compiler features
target_compile_features(test_video_recording PRIVATE cxx_std_11)

# Add definitions
target_compile_definitions(test_video_recording PRIVATE ${OpenCASCADE_COMPILE_DEFINITIONS})
